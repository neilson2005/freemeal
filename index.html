'Summary'; // ──────────────────────────────────────────────────────────── // ============================================================ // ENTRY POINT — called by the time-based trigger // ============================================================ function runPipeline() { const { startUtc, endUtc, label } = getYesterdayWindowUtc(); Logger.log(`Running pipeline for: ${label} 00:00:00 → 23:59:59 MYT (UTC+8) // ============================================================ function getYesterdayWindowUtc() { const now = new Date(); const myt = new Date(now.getTime() + 8 * 60 * 60 * 1000); // shift to MYT const yMyt = new Date(myt); yMyt.setUTCDate(yMyt.getUTCDate() - 1); const yDate = yMyt.toISOString().slice(0, 10); // YYYY-MM-DD MYT yesterday const startUtc = new Date(yDate + 'T00:00:00+08:00').toISOString().replace('.000', ''); const endUtc = new Date(yDate + 'T23:59:59+08:00').toISOString().replace('.000', ''); return { startUtc, endUtc, label: yDate }; } // ============================================================ // BIGQUERY — run the meal detail query // ============================================================ function queryBigQuery(startUtc, endUtc) { const sql = buildSql(startUtc, endUtc); const request = { query : sql, useLegacySql : false, timeoutMs : 120000, location : 'asia-southeast1' // adjust to your BQ dataset region }; let response = BigQuery.Jobs.query(request, GCP_PROJECT_ID); const jobId = response.jobReference.jobId; // Poll until complete let attempts = 0; while (!response.jobComplete && attempts < 30) { Utilities.sleep(3000); response = BigQuery.Jobs.getQueryResults(GCP_PROJECT_ID, jobId, { timeoutMs: 10000, location: 'asia-southeast1' }); attempts++; } if (!response.jobComplete) throw new Error('BigQuery job timed out'); const schema = response.schema.fields.map(f => f.name); const rows = []; let pageToken = null; do { const opts = { maxResults: 2000, location: 'asia-southeast1' }; if = r.f[i].v; }); rows.push(obj); }); pageToken = page.pageToken; } while (pageToken); return rows; } // ============================================================ // SQL — parameterised by UTC window // Mirrors the full meal detail query (deduped_meals CTE) // ============================================================ function buildSql(startUtc, endUtc) { return ` -- Free Meal Detail — parameterised daily run -- Window: booking created ${startUtc} → ${endUtc} (UTC) WITH meal_ssr_codes AS ( SELECT ssr_code, name AS ssr_name, ssr_type FROM NAVITAIRE.ssr WHERE ssr_type = 2 AND COALESCE(inactive, 0) = 0 ), first_leg_per_segment AS ( SELECT passenger_id, segment_id, MIN(inventory_leg_id) AS inventory_leg_id FROM NAVITAIRE.passenger_journey_leg GROUP BY passenger_id, segment_id ), meal_ssr_on_flights AS ( SELECT DISTINCT pjs.passenger_id, pjs.segment_id, pjs.booking_status, pjssr.ssr_code, pjssr.ssr_detail, pjssr.is_in_service_bundle, fl.inventory_leg_id FROM NAVITAIRE.passenger_journey_segment AS pjs INNER JOIN NAVITAIRE.passenger_journey_ssr AS pjssr ON pjs.passenger_id = pjssr.passenger_id AND pjs.segment_id = pjssr.segment_id INNER JOIN meal_ssr_codes AS mc ON pjssr.ssr_code = mc.ssr_code INNER JOIN first_leg_per_segment AS fl ON pjs.passenger_id = fl.passenger_id AND pjs.segment_id = fl.segment_id WHERE pjs.booking_status = 'HK' AND COALESCE(pjssr.delete_flag, 0) = 0 ), station_country AS ( SELECT station_code, country_code FROM NAVITAIRE.station ), meal_with_flight AS ( SELECT mof.passenger_id, mof.segment_id, mof.ssr_code, mof.ssr_detail, mof.is_in_service_bundle, io.carrier_code, io.flight_number, io.departure_station, io.arrival_station, dep_stn.country_code AS departure_country, arr_stn.country_code AS arrival_country, CASE WHEN dep_stn.country_code IS NULL OR arr_stn.country_code IS NULL THEN 'UNKNOWN' WHEN dep_stn.country_code = arr_stn.country_code THEN 'DOMESTIC' ELSE 'INTERNATIONAL' END AS flight_type, DATE(io.std_utc) AS flight_date, io.std_utc, io.sta_utc, TIMESTAMP_DIFF(io.sta_utc, io.std_utc, MINUTE) AS flight_duration_mins, CASE WHEN TIMESTAMP_DIFF(io.sta_utc, io.std_utc, MINUTE) < 75 THEN '< 75 mins' ELSE '>= 75 mins' END AS flight_duration_category, mc.ssr_name, mc.ssr_type FROM meal_ssr_on_flights AS mof INNER JOIN NAVITAIRE.inventory_object AS io ON mof.inventory_leg_id = io.inventory_leg_id INNER JOIN meal_ssr_codes AS mc ON mof.ssr_code = mc.ssr_code LEFT JOIN station_country AS dep_stn ON io.departure_station = dep_stn.station_code LEFT JOIN station_country AS arr_stn ON io.arrival_station = arr_stn.station_code WHERE DATE(io.std_utc) >= DATE_SUB(CURRENT_DATE('Asia/Kuala_Lumpur'), INTERVAL 90 DAY) AND DATE(io.std_utc) <= '2026-08-31' ), meal_with_booking AS ( SELECT b.record_locator, b.booking_id, CASE WHEN b.booking_promo_code = 'FREEMEAL' AND mwf.ssr_code IN ('CRCB','ORMX','UCTX','CLCB') AND mwf.flight_type = 'INTERNATIONAL' THEN 'FREEMEAL' ELSE NULL END AS booking_promo_code, b.created_date AS booking_created_date, DATETIME(b.created_date, 'Asia/Kuala_Lumpur') AS booking_created_date_my, bp.passenger_id, bp.first_name, bp.last_name, bp.pax_type, mwf.carrier_code, mwf.flight_number, mwf.flight_date AS departure_date, mwf.departure_station, mwf.arrival_station, mwf.departure_country, mwf.arrival_country, mwf.flight_type, mwf.std_utc, mwf.sta_utc, mwf.flight_duration_mins, mwf.flight_duration_category, mwf.ssr_code, mwf.ssr_detail, mwf.is_in_service_bundle, mwf.ssr_name, mwf.segment_id FROM meal_with_flight AS mwf INNER JOIN NAVITAIRE.booking_passenger AS bp ON mwf.passenger_id = bp.passenger_id INNER JOIN NAVITAIRE.booking AS b ON bp.booking_id = b.booking_id WHERE b.status = 2 AND b.created_date >= TIMESTAMP('${startUtc}') AND b.created_date < TIMESTAMP('${endUtc}') ), deduped_meals AS ( SELECT *, ROW_NUMBER() OVER ( PARTITION BY booking_id, passenger_id, flight_number, departure_date, ssr_code, booking_promo_code ORDER BY segment_id ) AS rn FROM meal_with_booking ) SELECT record_locator, booking_id, booking_promo_code, CAST(booking_created_date AS STRING) AS booking_created_date, CAST(booking_created_date_my AS STRING) AS booking_created_date_my, passenger_id, first_name, last_name, pax_type, carrier_code, flight_number, CAST(departure_date AS STRING) AS departure_date, departure_station, arrival_station, departure_country, arrival_country, flight_type, CAST(std_utc AS STRING) AS std_utc, CAST(sta_utc AS STRING) AS sta_utc, flight_duration_mins, flight_duration_category, ssr_code, ssr_detail, CAST(is_in_service_bundle AS STRING) AS is_in_service_bundle, ssr_name FROM deduped_meals WHERE rn = 1 ORDER BY departure_date, std_utc, carrier_code, flight_number, passenger_id, ssr_code `; } // ============================================================ // WRITE RAW DETAIL TAB // ============================================================ function writeRawDetail(rows, label) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); let sheet = ss.getSheetByName(RAW_SHEET); if (!sheet) sheet = ss.insertSheet(RAW_SHEET); const COLS = [ 'record_locator','booking_id','booking_promo_code', 'booking_created_date','booking_created_date_my', 'passenger_id','first_name','last_name','pax_type', 'carrier_code','flight_number','departure_date', 'departure_station','arrival_station', 'departure_country','arrival_country','flight_type', 'std_utc','sta_utc','flight_duration_mins','flight_duration_category', 'ssr_code','ssr_detail','is_in_service_bundle','ssr_name', 'pipeline_run_date', 'pipeline_run_timestamp' ]; if (sheet.getLastRow() === 0) { sheet.appendRow(COLS); sheet.getRange(1, 1, 1, COLS.length).setFontWeight('bold'); sheet.setFrozenRows(1); } if (rows.length === 0) { Logger.log(`RawDetail: no rows for ${label} — nothing appended.`); return; } const nowMyt = new Date(new Date().getTime() + 8 * 60 * 60 * 1000); const runTs = nowMyt.toISOString().replace('T', ' ').slice(0, 19); const data = rows.map(r => COLS.map(c => { if (c === 'pipeline_run_date') return label; if (c === 'pipeline_run_timestamp') return runTs; return r[c] ?? ''; })); sheet.getRange(sheet.getLastRow() + 1, 1, data.length, COLS.length).setValues(data); Logger.log(`RawDetail: appended ${rows.length} rows for ${label} at ${runTs}`); } // ============================================================ // WRITE SUMMARY TAB // ============================================================ function writeSummary(rows, label) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); let sheet = ss.getSheetByName(SUMMARY_SHEET); if (!sheet) sheet = ss.insertSheet(SUMMARY_SHEET); sheet.clearContents(); const output = []; // ── Metadata ──────────────────────────────────────────── output.push(['pipeline_run_date', label]); output.push(['total_ssr_count', rows.length]); output.push(['unique_flights', [...new Set(rows.map(r => r.carrier_code + r.flight_number + r.departure_date))].length]); output.push(['unique_sectors', [...new Set(rows.map(r => r.departure_station + '-' + r.arrival_station))].length]); output.push([]); // ── View 1: by date + duration ─────────────────────────── output.push(['[view1] by_date_duration']); output.push(['departure_date','flight_duration_category','booking_promo_code','ssr_count','unique_pax']); const v1 = groupBy(rows, r => `${r.departure_date}|${r.flight_duration_category}|${r.booking_promo_code}`); Object.entries(v1).forEach(([key, grp]) => { const [date, dur, promo] = key.split('|'); output.push([date, dur, promo, grp.length, new Set(grp.map(r=>r.passenger_id)).size]); }); output.push([]); // ── View 2: by carrier ─────────────────────────────────── output.push(['[view2] by_carrier']); output.push(['carrier_code','flight_duration_category','booking_promo_code','ssr_count','unique_pax']); const v2 = groupBy(rows, r => `${r.carrier_code}|${r.flight_duration_category}|${r.booking_promo_code}`); Object.entries(v2).forEach(([key, grp]) => { const [carrier, dur, promo] = key.split('|'); output.push([carrier, dur, promo, grp.length, new Set(grp.map(r=>r.passenger_id)).size]); }); output.push([]); // ── View 3: by sector ──────────────────────────────────── output.push(['[view3] by_sector']); output.push(['sector','departure_station','arrival_station','flight_type','flight_duration_category','booking_promo_code','ssr_count','unique_pax']); const v3 = groupBy(rows, r => `${r.departure_station}|${r.arrival_station}|${r.flight_type}|${r.flight_duration_category}|${r.booking_promo_code}`); Object.entries(v3).forEach(([key, grp]) => { const [dep, arr, ftype, dur, promo] = key.split('|'); output.push([`${dep}-${arr}`, dep, arr, ftype, dur, promo, grp.length, new Set(grp.map(r=>r.passenger_id)).size]); }); output.push([]); // ── View 4: by promo split ─────────────────────────────── output.push(['[view4] by_promo']); output.push(['booking_promo_code','flight_type','ssr_code','ssr_count','unique_pax']); const v4 = groupBy(rows, r => `${r.booking_promo_code}|${r.flight_type}|${r.ssr_code}`); Object.entries(v4).forEach(([key, grp]) => { const [promo, ftype, ssr] = key.split('|'); output.push([promo, ftype, ssr, grp.length, new Set(grp.map(r=>r.passenger_id)).size]); }); output.push([]); // ── View 5: by flight ──────────────────────────────────── output.push(['[view5] by_flight']); output.push(['carrier_code', 'flight_number', 'ssr_count', 'unique_pax']); const v5 = groupBy(rows, r => `${r.carrier_code}|${r.flight_number}`); Object.entries(v5).forEach(([key, grp]) => { const [carrier, flight] = key.split('|'); output.push([carrier, flight, grp.length, new Set(grp.map(r=>r.passenger_id)).size]); }); output.push([]); // Padding rows to ensure 8 columns width for correct rendering sheet.getRange(1, 1, output.length, Math.max(...output.map(r=>r.length))).setValues( output.map(r => { while(r.length < 8) r.push(''); return r; }) ); // Bold section headers [1,6,...findSectionRows(output)].forEach(r => { try { sheet.getRange(r, 1, 1, 8).setFontWeight('bold'); } catch(e){} }); Logger.log(`Summary: written for ${label}`); } // ============================================================ // HELPERS // ============================================================ function groupBy(arr, keyFn) { return arr.reduce((acc, item) => { const k = keyFn(item); if (!acc[k]) acc[k] = []; acc[k].push(item); return acc; }, {}); } function findSectionRows(output) { const rows = []; output.forEach((r, i) => { if (typeof r[0] === 'string' && r[0].startsWith('[view')) rows.push(i + 1); }); return rows; } // ============================================================ // TRIGGER SETUP // ============================================================ function setupTrigger() { ScriptApp.getProjectTriggers().forEach(t => { if (t.getHandlerFunction() === 'runPipeline') ScriptApp.deleteTrigger(t); }); ScriptApp.newTrigger('runPipeline') .timeBased() .everyDays(1) .atHour(9) .create(); Logger.log('Trigger set: runPipeline daily at 9am (script timezone).'); }
